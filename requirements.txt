# Minimal direct dependencies for syncswim-dashboard
# Lets pip resolve transitive deps to whatever's compatible on
# the target machine (Apple Silicon, Intel Mac, Linux all OK).
#
# Why minimal vs `pip freeze`:
#   - Tim 老师's source venv has packages from a custom pip index
#     (numpy 2.2.6, protobuf 6.33.6, mediapipe 0.10.33, torch 2.11.0
#     etc — none of these exist on standard PyPI). Freezing those
#     into requirements.txt makes the repo unusable on any machine
#     with vanilla PyPI.
#   - mediapipe 0.10.x requires numpy<2 — pinning numpy==2.x guarantees
#     a resolution conflict on every PyPI install.
#
# This list = only top-level packages we directly import. Transitive
# deps (pandas, scipy, contourpy, torch, torchvision, etc) come along
# automatically at versions pip considers compatible.

# ──────────────── Web framework ────────────────
fastapi
uvicorn[standard]
aiofiles
websockets

# ──────────────── Config / data ────────────────
pydantic>=2.0
tomli; python_version < "3.11"

# ──────────────── BLE ────────────────
bleak

# ──────────────── Computer vision ────────────────
# numpy<2 is REQUIRED — mediapipe 0.10.x's wheel was built against
# numpy 1.x and breaks if numpy 2.x is in the env. ultralytics +
# opencv all work fine with numpy 1.x too.
numpy>=1.23,<2
opencv-python>=4.8.0
opencv-contrib-python>=4.8.0
pillow>=10.0
mediapipe>=0.10.5,<0.11

# ──────────────── YOLO (auto-pulls torch + torchvision) ────────────────
ultralytics>=8.3.0

# ──────────────── Reports / analysis ────────────────
matplotlib>=3.7
scipy>=1.10

# ──────────────── Testing ────────────────
pytest>=7.0
httpx>=0.25
